hugel,178,142,0	script	toairplane#hugel	45,1,1,{
	end;
OnTouch:
	mes "Para usar o dirigível, que são obrigados a pagar 1.200 zeny ou de um Bilhete Airship grátis.";
	mes "Gostaria de usar o serviço?";
	next;
	if (select("Sim:Não") == 1)
	{
		if (countitem(7311) > 0)
		{
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane",244,58;
			end;
		}
		if (Zeny >= 1200)
		{
			set Zeny, Zeny - 1200;
			warp "airplane",244,58;
			end;
		}
		mes "Lamento, mas você não tem dinheiro suficiente.";
		mes "Por favor, lembre-se, você é obrigado a pagar 1.200 zenys para usar o serviço.";
		close;
	}
	mes "Obrigado, por favor, venha novamente.";
	close;
}
