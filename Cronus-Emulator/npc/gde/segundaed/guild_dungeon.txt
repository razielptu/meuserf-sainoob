schg_cas01,280,391,3	script	Girassol misterioso#01	976,{
	set .@GID, GetCastleData("schg_cas01",1);
	if (getcharid(2) == .@GID) {
		mes "-É um girassol incrivelmente enorme, tão grande quanto um ser humano! ... Você sente algo misterioso que emana da flor. -";
		next;
		switch(select("Mantenha o tronco:Fazer nada.")) {
		case 1:
			warp "schg_dun01",262,314;
			end;
		case 2:
			mes "É muito assustador para tocar coisas desconhecidas.";
			close;
		}
	}
}
