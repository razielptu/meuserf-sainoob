alberta,245,66,4	script	Oficial de Moscóvia#1	960,{
	mes "[Oficial de Moscóvia]";
	mes "Moscóvia! Você conhece Moscóvia?";
	mes "O paraíso se espalhando nos mares sem fim...";
	mes "Bem-vindo a Moscóvia";
	mes "É uma terra mística.";
	next;
	switch(select("Sobre Moscóvia...:Ir para Moscóvia.:Cancelar")) {
		case 1:
			mes "[Oficial de Moscóvia]";
			mes "Moscóvia é um lindo reino";
			mes "uma Ilha localizada no norte";
			mes "de Rune-Midgarts.";
			next;
			mes "[Oficial de Moscóvia]";
			mes "Sei que você ficará";
			mes "absolutamente fascinado";
			mes "cenário de Moscóvia";
			mes "e pelo belo Palácio.";
			next;
			mes "[Oficial de Moscóvia]";
			mes "Agora que nosso longo inverno";
			mes "passou, estou feliz";
			mes "por poder mostrar a você";
			mes "nossa linda cidade.";
			next;
			mes "[Oficial de Moscóvia]";
			mes "Se você quiser visitar";
			mes "Moscóvia, vá agora!";
			mes "Eu levarei você em uma boa viagem";
			mes "por Moscóvia!";
			close;
			
		case 2:
			mes "[Oficial de Moscóvia]";
			mes "Ok, vamos começar agora.";
			mes "Você deve me pagar 10.000 zenys";
			mes "para ir á Moscóvia.";
			mes "Mas não precisa pagar";
			mes "para voltar.";
			next;
			mes "[Oficial de Moscóvia]";
			mes "Podemos ir?";
			next;
			if (select("Vamos!:Cancelar") == 2) {
				mes "[Oficial de Moscóvia]";
				mes "Se você está ocupado agora,";
				mes "me avise quando quiser ir.";
				mes "Estou sempre pronto para guiar alguém por Moscóvia.";
				close;
			}
			if (Zeny < 10000) {
				mes "[Oficial de Moscóvia]";
				mes "Sinto muito, mas você";
				mes "não tem dinheiro suficiente.";
				mes "Precisa de 10 000 zenys";
				mes "para ir a Moscóvia.";
				mes "Obrigado.";
				close;
			}else{
				mes "[Oficial de Moscóvia]";
				mes "Certo, estamos indo agora.";
				close2;
				set Zeny,Zeny - 10000;
				warp "moscovia",163,55;
				end;
			}
		case 3:
			mes "[Oficial de Moscóvia]";
			mes "Se você está ocupado agora,";
			mes "por favor me avise quando quiser ir.";
			mes "Estou sempre pronto para guia-lo por Moscóvia.";
			close;
	}
}
