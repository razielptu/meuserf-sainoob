function	script	EinTower	{
	mes "[" + getarg(0) + "]";
	mes "Bom-dia~";
	mes "Sou" + getarg(0) + ", seu";
	mes "guia aqui na";
	mes "Torre de Einbroch.";
	next;
	mes "[" + getarg(0) + "]";
	mes "A Torre de Einbroch oferece";
	mes "a melhor vista da nossa cidade,";
	mes "e é um ótimo lugar para";
	mes "encontros com amigos ou";
	mes "pretendentes. O ingresso para";
	mes "a Torre custa apenas 10 zenys.";
	next;
	mes "[" + getarg(0) + "]";
	mes "Atualmente, estamos oferecendo";
	mes "uma promoção especial";
	mes "chamada Set Combo da Maçã,";
	mes "por 20 zenys. Esse set incluí";
	mes "o ingresso para a Torre e uma";
	mes "maçã para você fazer um lanchinho!";
	next;
	switch(select("Ingresso para a Torre:Set Combo da Maçã:Cancelar")) {
	case 1:
		if (Zeny < 10) {
			mes "[" + getarg(0) + "]";
			mes "Sinto muito, mas";
			mes "você não tem zenys";
			mes "suficientes. O ingresso";
			mes "para a Torre de Einbroch";
			mes "custa 10 zenys.";
			close;
		}
		else {
			mes "[" + getarg(0) + "]";
			mes "Obrigado por utilizar";
			mes "nossos serviços.";
			mes "Deixe-me guiá-lo";
			mes "para a torre agora.";
			set zeny,zeny-10;
			close2;
			warp "einbroch",181,196;
			end;
		}
	case 2:
		if (Zeny < 20) {
			mes "[" + getarg(0) + "]";
			mes "Sinto muito, mas você não tem";
			mes "zenys suficientes. O Set Combo";
			mes "da Maçã custa 20 zenys.";
			close;
		}
		else {
			mes "[" + getarg(0) + "]";
			mes "Antes de guiá-lo até a torre,";
			mes "vou verificar seu status";
			mes "para garantir a sua segurança";
			mes "antes de eu entregar a você";
			mes "o Set Combo da Maçã.";
			next;
			if (checkweight(512,1)) {
				mes "[" + getarg(0) + "]";
				mes "Obrigado por utilizar";
				mes "os nossos serviços.";
				mes "Deixe-me guiá-lo até";
				mes "a torre agora.";
				set zeny,zeny-20;
				getitem 512,1; //Apple
				close2;
				warp "einbroch",174,204;
				end;
			}
			else {
				mes "[" + getarg(0) + "]";
				mes "Sinto muito, mas você está carregando";
				mes "itens demais. Deposite algumas coisas antes de comprar";
				mes "o Set Combo da Maçã.";
				close;
			}
		}
	case 3:
		mes "[" + getarg(0) + "]";
		mes "Entendo.";
		mes "Volte quando";
		mes "quiser.";
		mes "Obrigado.";
		close;
	}
}
