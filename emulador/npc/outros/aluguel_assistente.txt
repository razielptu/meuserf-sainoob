//prontera,41,337,5	script	Gerente do Clã de Assis::Spear	105,{
	set .@faith_lancer, mercenary_get_faith(SPEAR_MERC_GUILD);
	show "[Gerente de Assistentes]";
	show "Bem-vindo à";
	show "Guilda dos Assistentes Lanceiros.";
	show "Em que posso ajudar?";
	next;
	switch( select("Contratar Assistente:Informações sobre Assistentes:Nada:Assistentes de 10º Nível") )
	{
	case 1:
		show "[Gerente de Assistentes]";
		show "Você deseja contratar um Assistente Lanceiro?";
		show "Qual Nível de Assistente você gostaria de contratar?";
		next;
		set .@Grade, select("1º Nível:2º Nível:3º Nível:4º Nível:5º Nível:6º Nível:7º Nível:8º Nível:9º Nível");
		set .@BaseLevel, 5 + (.@Grade * 10);
		if( .@BaseLevel > 90 )
			set .@BaseLevel, 90;

		set .@ZenyCost, 10 * .@Grade;

		if( .@Grade == 7 )
			set .@FaithCost, 50;
		else if( .@Grade == 8 )
			set .@FaithCost, 100;
		else if( .@Grade == 9 )
			set .@FaithCost, 300;
		else
			set .@FaithCost, 0;

		show "[Gerente de Assistentes]";
		show "Então você quer contratar um Assistente";
		show "Lanceiro de " + .@Grade + "º Nível?";
		show "Você precisa ter";
		show "Nível de Base " + .@BaseLevel + " ou maior, além de";
		show "pagar a taxa de serviço de " + .@ZenyCost + ",000 zenys.";
		next;
		if( select("Sim:Não") == 2 )
		{
			show "[Gerente de Assistentes]";
			show "Oh, é mesmo? Bem, agora";
			show "talvez não seja uma boa hora";
			show "para você pensar em contratar";
			show "um Assistente, mas sinta-se à vontade";
			show "para voltar caso mude";
			show "de idéia. Obrigado.";

		}
		else if( .@FaithCost && .@faith_lancer < .@FaithCost )
		{
			show "[Gerente de Assistentes]";
			show "Sinto muito, mas sua";
			show "taxa de Lealdade é muito";
			show "baixa para contratar um Assistente";
			show "de " + .@Grade + "º Nível.Você deve ter";
			show .@FaithCost+" ou mais pontos de Lealdade para";
			show "fazer um contrato com um deles.";
		}
		else if( BaseLevel < .@BaseLevel )
		{
			show "[Gerente de Assistentes]";
			show "Sinto muito, mas seu Nível";
			show "Base não é alto o suficiente";
			show "para contratar este Assistente.";
			show "Por favor, volte";
			show "assim que tiver atingido o Nível Base " + .@BaseLevel + ".";

		}
		else if( Zeny < .@ZenyCost * 1000 )
		{
			show "[Gerente de Assistentes]";
			show "Sinto muito, mas";
			show "você não tem zenys";
			show "para contratar este Assistente.";
			show "A taxa de contratação é de " + .@ZenyCost + ",000 zenys.";
		}
		else
		{
			show "[Gerente de Assistentes]";
			show " Ótimo! Nossos Assistentes";
			show "são honestos e devotados";
			show "à proteção de nossos clientes.";
			show "Os Assistentes convocados";
			show "oferecerão seus serviços a você por";
			show "30 minutos. Cuide-se.";
			set Zeny, Zeny - (.@ZenyCost * 1000);
			additem 12172 + .@Grade, 1;
		}
		close;
	case 2:
		show "[Gerente de Assistentes]";
		show "Assistentes são soldados";
		show "que lutarão a seu lado";
		show "no campo de batalham, mas existem";
		show "alguns termos e condições";
		show "que você deve cumprir para";
		show "contratá-los.";
		next;
		show "[Gerente de Assistentes]";
		show "Você deve cumprir o requisito";
		show "de nível e pagar uma taxa em zenys";
		show "para contratar um Assistente.";
		show "Assistente de alto nível";
		show "também têm como condição que você";
		show "tenha um certo nível de Lealdade a nós.";
		next;
		show "[Gerente de Assistentes]";
		show "Contratos de Assistentes não podem";
		show "ser transferidos para outras pessoas,";
		show "e nós só permitimos uma diferença";
		show "de 5 Níveis Base entre Assistente";
		show "e cliente, então você não pode";
		show "contratar alguém muito mais forte que você.";
		next;
		show "[Gerente de Assistentes]";
		show "Bem, você pode se informar melhor dos";
		show "detalhes quando você de fato";
		show "contratar um de nossos";
		show "Assistentes e receber";
		show "o Pergaminho de Invocação, que irá";
		show "convocar um Assistente para lhe escoltar.";
		next;
		show "[Gerente de Assistentes]";
		show "Você não pode dar esse pergaminho";
		show "para outra pessoa e o";
		show "Assistente só ficará";
		show "com você até 30 minutos depois";
		show "que você convocá-lo. Não esqueça";
		show "do limite de tempo, ok?";
		close;
	case 3:
		show "[Gerente de Assistentes]";
		show "Não? Você não precisa de nenhuma";
		show "ajuda? Bem, fique à vontade para";
		show "perguntar o que quiser";
		show "saber sobre Assistentes.";
		close;
	case 4:
		show "[Gerente de Assistentes]";
		show "Assistentes de 10º Nível são";
		show "os melhores que temos,";
		show "por isso usamos outros critérios";
		show "para nossos clientes contratá-los";
		show "Não há taxa de zeny, mas você";
		show "deve ter 500 pontos de Lealdade à Guilda.";
		next;
		show "[Gerente de Assistentes]";
		show "Ao fazer um contrato";
		show "com um Assistente de 10º Nível,";
		show "sua taxa de Lealdade será";
		show "reduzida em 400 pontos. Ou seja,";
		show "você paga 400 pontos de Lealdade";
		show "para contratar um Assistente de 10º Nível.";
		next;
		show "[Gerente de Assistentes]";
		show "Você também deve ter";
		show "Nível Base 90 ou maior para";
		show "contratar um Assistente de 10º Nível.";
		show "Você ainda está interessado em fazer este contrato?";
		next;
		if( select("Sim:Não") == 2 )
		{
			show "[Gerente de Assistentes]";
			show "Entendo... É necessário";
			show "muito esforço e sacrifício para";
			show "alcançar os requisitos";
			show "de contrato de um Assistente";
			show "de 10º Nível. Você já pensou em";
			show "contratar um Assistente de nível menor?";

		}
		else if( .@faith_lancer < 500 )
		{
			show "[Gerente de Assistentes]";
			show "Sinto muito, mas sua";
			show "taxa de Lealdade é muito";
			show "baixa para contratar um Assistente";
			show "de 10º Nível.Você deve ter";
			show "500 ou mais pontos de Lealdade para";
			show "fazer um contrato com um deles.";

		}
		else if( BaseLevel < 90 )
		{
			show "[Gerente de Assistentes]";
			show "Sinto muito, mas seu Nível";
			show "Base não é alto o suficiente";
			show "para contratar este Assistente.";
			show "Por favor, volte";
			show "assim que tiver atingido o Nível Base 90.";

		}
		else
		{
			show "[Gerente de Assistentes]";
			show " Ótimo! Nossos Assistentes";
			show "são honestos e devotados";
			show "à proteção de nossos clientes.";
			show "Os Assistentes convocados";
			show "oferecerão seus serviços a você por";
			show "30 minutos. Cuide-se.";

			mercenary_set_faith SPEAR_MERC_GUILD,-400;
			additem 12182,1;
		}
		close;
	}
}
