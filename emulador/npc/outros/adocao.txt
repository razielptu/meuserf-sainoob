function	script	AdoptReset	{
	deltimer "TimerAdopt::OnFinishTimer";
	set $@AdoptionActive, 0;
	set $@ParentOne$, null;
	set $@ParentTwo$, null;
	set $@Baby$, null;
	set $@GenderBaby$, null;
	set $@FinalQuestion, 0;
	set $@CheckAll, 0;
	areaannounce "prt_church",0,0,350,350,"A adoção foi cancelada por causa "+getarg(0),0;
	return;
}
