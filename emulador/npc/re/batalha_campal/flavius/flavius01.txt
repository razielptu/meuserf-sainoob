	end;

			set $@Croix_ScoreBG1,1;
			donpcevent "time#bat_b01::OnEnable";
			donpcevent "start#bat_b01::onReset";
		}

		bg_updatescore "bat_b01",$@Guill_ScoreBG1,$@Croix_ScoreBG1;

		bg_warp $@FlaviusBG1_id2,"bat_b01",390,10;

	end;
