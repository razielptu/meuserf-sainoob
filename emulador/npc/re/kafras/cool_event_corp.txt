hugel,88,168,4	script	Corporação de Eventos::StaffHugel	874,{
	cutin "zonda_01",2;
	mes "[Corporação de Eventos Incríveis]";
	mes "Bem-vindo a Corporação de Eventos incríveis.";
	mes "Nós estamos sempre trabalhando";
	mes "para superar suas expectativas";
	mes "de um serviço de qualidade. Como";
	mes "posso lhe ser útil?";
	next;
	switch(select("Salver:Usar armazém:Alugar Carrinho:Senha do Armazém:Cancelar")){
	case 1:
		mes "[Corporação de Eventos Incríveis]";
		mes "Seu ponto de retorno";
		mes "foi salvo aqui, na vila";
		mes "de Hugel. Obrigado por";
		mes "usar o serviço da Corporação";
		mes "de Eventos Incríveis~";
		savepoint "hugel",96,155;
		close2;
		break;
	case 2:
		mes "[Corporação de Eventos Incríveis]";
		if(basicskillcheck() && getskilllv("NV_BASIC") < 6){
			mes "Sinto muito, mas você";
			mes "precisa de Habilidades";
			mes "Básicas nível 6 para";
			mes "usar o serviço de armazém.";
			close2;
			break;
		}
		if(Zeny < 40){
			mes "Desculpe, mas você não";
			mes "possui zeny suficiente para";
			mes "usar o serviço de armazém. Nossa";
			mes "taxa de armazém é de 40 zeny.";
			close2;
			break;
		}
		set Zeny, Zeny - 40;
		mes "Obrigador por apoiar a";
		mes "Corporação de Eventos Incríveis.";
		mes "Deixe-me abrir seu";
		mes "armazém pessoal imediatamente";
		mes "Tenha um ótimo dia~";
		callfunc("F_CheckKafCode");	//checa a senha do seu armazém, caso a mesma exista
		close2;
		openstorage;
		break;
	case 3:
		mes "[Corporação de Eventos Incríveis]";
		if(BaseClass != Job_Merchant){
			mes "Sinto muito, mas";
			mes "o serviço de aluguél de carrinhos";
			mes "está disponível apenas para Mercadores,";
			mes "Ferreiros, Mestres-Ferreiros,";
			mes "Alquimistas e Criadores.";
			close2;
			break;
		}
		if(getskilllv("MC_PUSHCART") == 0){
			mes "Você só pode alugar um carrinho";
			mes "após aprender a habilidade ''Usar Carrinho''.";
			close2;
			break;
		}
		if(checkcart() == 1){
			mes "Você já possui";
			mes "um carrinho alugado.";
			mes "Infelizmente, não podemos";
			mes "alugar mais de um carrinho";
			mes "simultâneamente para um cliente.";
			close2;
			break;
		}
		if(countitem(7061)){
			delitem 7061,1;
			setcart;
			close2;
			break;
		}
		mes "O aluguél de carrinhos";
		mes "custa 800 zeny. Você deseja";
		mes "alugar um carrinho?";
		next;
		if(select("Alugar um Carrinho:Cancelar") == 1){
			if(Zeny < 800){
				mes "[Corporação de Eventos Incríveis]";
				mes "Sinto muito, mas";
				mes "você não possui a quantidade";
				mes "de zenys necessária para";
				mes "o serviço de aluguél de carrinhos.";
			} else {
				set Zeny, Zeny - 800;
				setcart;
			}
		}
		close2;
		break;
	case 4:
		callfunc("F_SetKafCode","[Corporação de Eventos Incríveis]","Corporação de Eventos Incríveis.");
		//a função fecha a caixa de mensagem e remove todos os Cutins
	case 5:
	default:
		mes "[Corporação de Eventos Incríveis]";
		mes "Corporação de Eventos Incríveis está sempre";
		mes "procurando a melhor maneira de apoiar";
		mes "os aventureiros de Rune Midgard.";
		mes "Nos ajude a nos tornarmos os melhores";
		mes "fornecendo-nos sua opinião";
		mes "honesta e sincera. Obrigado.";
		close2;
	}
	cutin "", 255;
	end;
}
