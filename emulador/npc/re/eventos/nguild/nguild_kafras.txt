function	script	F_GKafra	{
	cutin "kafra_01",2;
	set @GID, GetCastleData(getarg(0),1);
	if(getcharid(2) == @GID && getgdskilllv(@GID,10001)){
		set @wrpP[0], 200;
		set @wrpD$[0], getarg(1);
		setarray @wrpC$[0], @wrpD$[0]+" ^880000"+@wrpP[0]+"^000000 z", "Cancel", "", "", "","";
		callfunc "F_Kafra",2,0,0,0,800;
		end;
	}
	mes "[Kafra Recepcionista]";
	mes "Eu fui contratada pelo Clã ^ff0000" + getguildname(.@GID) + "^000000 e só posso atendê-los. Desculpe-me pelo inconveniente.";
	cutin "",255;
	close;
}
