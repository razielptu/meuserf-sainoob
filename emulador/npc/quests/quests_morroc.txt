	show "E eu tenho recebido muitos pedidos de ^3355FF'Placas de PARE'^000000 recentemente ;]";
	next;

	M_Order:
		show "[William]";
		show "A ^3355FF'Placa de PARE'^000000 cabe confortavelmente na cabeça das pessoas.";
		show "Não é somente um acessório interessante, como também está na moda.";
		next;
		show "[William]";
		show "Eu garanto que a nossa ^3355FF'Placa de PARE'^000000 dura bem mais que 100 anos, ou então não me chamo MacWillian!!";
		next;
		show "[William]";
		show "Para produzir a ^3355FF'Placa de PARE'^000000, você precisa:";
		show "^3355FF91100 Zeny,";
		show "50 Troncos,";
		show "e 1 Tintura Preta^000000.";
		next;
		if((countitem(1019) >= 50) && (countitem(983) >= 1) && (Zeny >= 91100)) goto sL_GetPost;
		show "[William]";
		show "Você não tem os itens necessários, volte mais tarde. ";
		close;

		sL_GetPost:
			show "[William]";
			show "Aqui está a ^3355FF'Placa de PARE '^000000! Obrigado.";
			additem 2272,1;
			delitem 1019,50;
			delitem 983,1;
			set Zeny, Zeny - 91100;
			emotion 15;
			close;

	M_End:
		show "[William]";
		show "Obrigado por parar na casa dos MacWillian's, até mais. ";
		close;
}
