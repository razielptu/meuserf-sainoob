prt_in,84,98,3	script	Esope	734,{
	show "[Esope]";
	if(BaseJob!=Job_Knight) goto L_noknight;
	if(getskilllv(1001)) goto L_alreadyhave;

	show "Eu poderei ensina-lo tudo que eu sei se você me trouxer:";
	show "5 Bengalas Doce";
	show "3 Rosas Eterna";
	next;
	
	show "[Esope]";
	show "Me deixe conferir seus items.";
	next;
	
	if(countitem(530)<5 || countitem(748)<3)goto L_noitems;//Items: Bengala_Doce, Rosa_Eterna,
	delitem 530, 5;//Items: Bengala Doce,
	delitem 748, 3;//Items: Rosa Eterna,
	
	show "[Esope]";
	show "Eu tenho o que você precisa,";
	show "então vou ensinar o meu talento.";
	skill 1001,1,0;
	close;

L_noitems:
	show "[Esope]";
	show "Você não possui os items que eu preciso.";
	show "Volte quando tiver todos os items exigidos.";
	close;
		
L_noknight:
	show "Procurando seu mestre?";
	emotion e_gasp;
	close;

L_alreadyhave:
	show "Eu já o ensinei";
	show "tudo que eu sei...";
	close;
}
