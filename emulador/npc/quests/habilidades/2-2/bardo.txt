prontera,134,328,7	script	Homem tímido	89,{
	show "[Homem tímido]";
	if(BaseJob!=Job_Bard) goto L_nobard;
	if(getskilllv(1010)) goto L_alreadyhave;
	if(JobLevel<40) goto L_nojob;

	show "Eu posso ensinar tudo que eu sei, mas eu quero que você me traga.";
	show "1 Tropical Sograt";
	next;
	
	show "[Homem tímido]";
	show "Me deixe conferir.";
	next;
	
	if (countitem(12112)<1) goto L_noitems;//Items: Tropical_Sograt,
	delitem 12112, 1;//Items: Tropical_Sograt,
	show "[Homem tímido]";
	show "Bem, to vendo que você trouxe o meu item, então vou ensina minha técnica.";
	skill 1010,1,0;
	close;
	
L_noitems:
	show "[Homem tímido]";
	show "Você não possui os items que eu preciso.";
	show "Volte quando tiver todos os items exigidos.";
	close;
		
L_nobard:
	show "Hum... você não parece ser um músico.";
	emotion e_gasp;
	close;

L_alreadyhave:
	show "Eu já o ensinei";
	show "tudo que eu sei...";
	close;

L_nojob:
	show "Volte quando você for um Bardo.";
	close;
}
