juperos_02,1,1,1	script	JuperosLvl2QuestInit	-1,{
OnInit:
while( 0 )
	{

		monster "juperos_02",24,276,"Interruptor da Porta 1",1674,1,"JuperosLvl2QuestInit::OnGate1Dead";
		monster "juperos_02",240,29,"Interruptor da Porta 2",1674,1,"JuperosLvl2QuestInit::OnGate2Dead";
		monster "juperos_02",282,183,"Interruptor da Porta 3",1674,1,"JuperosLvl2QuestInit::OnGate3Dead";
		end;

OnGate1Dead:

	mapannounce "juperos_02","Você está procurando por Juperos? Isto não existe...",1,0xFF0000;
	set $JupeGate1,1;
	if($JupeGate1 && $JupeGate2 && $JupeGate3)goto s_WarpOpen;
	end;

OnGate2Dead:

	mapannounce "juperos_02","Quem é você para vir aqui?",1,0xFF0000;
	set $JupeGate2,1;
	if($JupeGate1 && $JupeGate2 && $JupeGate3)goto s_WarpOpen;
	end;

OnGate3Dead:

	mapannounce "juperos_02","Você veio para me ver? Bom! Encontre-me primeiro!",1,0xFF0000;
	set $JupeGate3,1;
	if($JupeGate1 && $JupeGate2 && $JupeGate3)goto s_WarpOpen;
	end;

s_WarpOpen:
	attachnpctimer;
	initnpctimer;
		end;
	OnTimer3000:
		mapannounce "juperos_02","Vroom! Vroom!",1,0xBA55D3;
		end;
	OnTimer6000:
		mapannounce "juperos_02","Atenção, visitantes.",1,0xBA55D3;
		end;
	OnTimer9000:
		mapannounce "juperos_02","Você tem permissão de entrar na próxima zona por um curto período de tempo. Por favor, utilize o portal no centro do mapa.",1,0xBA55D3;
		enablenpc "JupeWarp2";
		end;
	OnTimer309000:
		mapannounce "juperos_02","O portal acaba de se fechar.",1,0xBA55D3;
		disablenpc "JupeWarp2";
		set $JupeGate1,0;
		set $JupeGate2,0;
		set $JupeGate3,0;
	stopnpctimer;
	}
}
