	show "[Phelix]";
	show "Hmm... você quer trocar jellopy por Poção Vermelha ou Batata Doce. O que quer?"; 
	next;
	switch(select("Poção Vermelha.","Batata Doce.")) {
	case 1:
		show "[Phelix]";
		show "Ok então deixe-me ver o que trouxe..."; 
		next;
		show "[Phelix]";
		if (countitem(909) < 10) {
			show "Ei! Não é o suficiente. Eu disse 1 Poção Vermelha por 10 Jellopy. Entendeu?";
			close;
		} else {
			set .@max,countitem(909)/10;
			show "Nada mal...";
			show "Quantas poções quer?";
			next;
			switch(select("Quantas eu tiver direito.","Te direi.","Nenhuma.")) {
			case 1:
				delitem 909,.@max*10;
				getitem 501,.@max;
				show "[Phelix]";
				show "Aqui estão! Poções como prometido.";
				show "Use-as sabiamente.";
				close;
