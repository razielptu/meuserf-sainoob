//yuno,141,175,5	script	Assistente do Aeroplano::1_yuno	67,{
	show "[Assistente do Aeroplano]";
	show "Bem vindo ao Aeroplano.";
	show "Em que posso ajudar?";
	next;
	switch(select("Embarcar no Aeroplano.:Cancelar.")){
		case 1:
			show "[Assistente do Aeroplano]";
			show "O bilhete para o Aeroplano custa 1.200 zenys. Se tiver um bilhete para o Aeroplano, pode usar este serviço de graça. Este vôo vai até ^9D283BHugel^000000. Quer embarcar?";
			next;
			if(select("Sim.:Não")==1)
				callsub S_WARP,"airplane",224,64;
			else
				break;
		case 2:
			break;
	}
	show "[Assistente do Aeroplano]";
	show "Volte sempre. Obrigado.";
	close;
S_WARP:
	if (Zeny >= 1200) {
		set zeny,zeny-1200;
		warp getarg(0),getarg(1),getarg(2);
		end;
	}
	else {
		show "[Assistente do Aeroplano]";
		show "Você não tem zeny suficiente para o serviço.";
		close;
	}
}
