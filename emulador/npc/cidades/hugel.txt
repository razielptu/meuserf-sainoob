	show "[Vendedor]";
	show "Bem-vindo à loja de artigos para";
	show "festas!";
	show "Por que não se divertir com os amigos";
	show "assistindo a uma espetacular queima";
	show "de fogos?";
	show "Vendemos 5 unidades por 500 zenys.";
	next;
	switch (select("Comprar:Cancelar")) {
		case 1:
			if (Zeny < 500) { 
				show "[Vendedor]";
				show "Desculpe, mas você não possuí";
				show "dinheiro suficiente.";
				close;
			}
			set Zeny,Zeny-500;
			additem 12018,5;
			show "[Vendedor]";
			show "Agora vá se divertir com seus fogos!";
			close;
		case 2:
			show "[Vendedor]";
			show "Obrigado, por favor volte sempre.";
			close;
	}
		
}
