// Função Loteria
function	script	F_Lottery	{
	show "[Kafra Recepcionista]";
	show "Este é o momento certo para tentar a sorte!!";
	next;
	show "[Kafra Recepcionista]";
	show "Tudo certo! Querido cliente, se prepare!!";
	next;
	show "'dum dum dum...'";
	next;
	digit @input;
	if(@input < 1 || @input > 5) set @input, rand(1,5); //Lupus's fix
	callsub sF_Spin;
	show "[Kafra Recepcionista]";
	show "Vamos lá... Você está pronto?";
	show "Vamos ver o resultado...";
	next;
	show "[Kafra Recepcionista]";
	show "^FF0000Uau! É um...!!^000000";
	next;
	if(@temp <  1) goto sL_Prize1;
	if(@temp <  2) goto sL_Prize2;
	if(@temp <  3) goto sL_Prize3;
	if(@temp <=  4) goto sL_Prize4;
	goto sL_Prize5;

	sL_Prize1:
		show "[Kafra Recepcionista]";
		show "1º lugar! 1º lugar!!";
		show "Você vai receber um presente pelo";
		show "2º lugar.";
		if(@Lotto == 1) additem 2328,1;//Items: Wooden_Mail,
		if(@Lotto == 2) additem 2307,1;//Items: Mantle,
		if(@Lotto == 3) additem 657,10;//Items: Berserk_Potion,
		if(@Lotto == 4) {
			additem 607,2;//Items: Yggdrasilberry,
			additem 608,1;//Items: Yggdrasil_Seed,
		}
		if(@Lotto == 5) additem 607,3;//Items: Yggdrasilberry,
		return;
	sL_Prize2:
		show "[Kafra Recepcionista]";
		show "2º lugar! 2º lugar!!";
		show "Você vai receber um presente pelo";
		show "2º lugar.";
		if(@Lotto == 1) additem 2403,1;//Items: Shoes,
		if(@Lotto == 2) additem 2226,1;//Items: Cap,
		if(@Lotto == 3) additem 2201,1;//Items: Sunglasses,
		if(@Lotto == 4) additem 526,3;//Items: Royal_Jelly,
		if(@Lotto == 5) {
			additem 608,1;//Items: Yggdrasil_Seed,
			additem 526,10;//Items: Royal_Jelly,
		}
		return;
	sL_Prize3:
		show "[Kafra Recepcionista]";
		show "3º lugar! 3º lugar!!";
		show "Você vai receber um presente pelo";
		show "3º lugar.";
		if(@Lotto == 1) additem 602,4;//Items: Butterfly_Wing,
		if(@Lotto == 2) additem 505,3;//Items: Blue_Potion,
		if(@Lotto == 3) additem 2203,1;//Items: Glasses,
		if(@Lotto == 4) additem 504,15;//Items: White_Potion,
		if(@Lotto == 5) additem 504,30;//Items: White_Potion,
		return;
	sL_Prize4:
		show "[Kafra Recepcionista]";
		show "4º lugar! 4º lugar!!";
		show "Você vai receber um presente pelo";
		show "4º lugar.";
		if(@Lotto == 1) additem 516,100;//Items: Sweet_Potato,
		if(@Lotto == 2) additem 501,150;//Items: Red_Potion,
		if(@Lotto == 3) additem 502,150;//Items: Orange_Potion,
		if(@Lotto == 4) additem 505,5;//Items: Blue_Potion,
		if(@Lotto == 5) additem 505,10;//Items: Blue_Potion,
		return;
	sL_Prize5:
		show "[Kafra Recepcionista]";
		show "5º lugar! 5º lugar!!";
		show "Você vai receber um presente pelo";
		show "5º lugar.";
		if(@Lotto == 1) additem 516,50;//Items: Sweet_Potato,
		if(@Lotto == 2) additem 501,100;//Items: Red_Potion,
		if(@Lotto == 3) additem 501,200;//Items: Red_Potion,
		if(@Lotto == 4) additem 501,250;//Items: Red_Potion,
		if(@Lotto == 5) additem 501,300;//Items: Red_Potion,
		return;

sF_Spin:
	show "[Máquina de Loteria]";
	show "Número de giros que você tem: "+@input;
	next;
	show "[Máquina de Loteria]";
	show "'dum dum dum...'";
	next;
	set @temp, rand(10);
	set @input, @input -1;
	if(@input <= 0) return;
	goto sF_Spin;
}
