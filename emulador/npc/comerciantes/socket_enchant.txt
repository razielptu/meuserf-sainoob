				show "Eu queria continuar, mas você está esquecendo de alguma coisa.";
				show "Tem certeza de que você está com o equipamento, com o zeny e com os itens?";
				close;
			}
			break;
	}
	
	show "[Seiyablem]";
	show "Tudo bem então, vamos começar!";
	show "Você devia rezar pra dar certo.";
	next;
	set .@socketrand,rand(1,100);
	if((.@socketrand > getarg(3)) && (.@socketrand < getarg(4)))
	{
		if(getarg(4) == 51) specialeffect EF_LORD;
		else specialeffect EF_SANCTUARY;
		show "[Seiyablem]";
		show "Ótimo, parece que vai dar certo.";
		show "Parece perfeito. Parabéns!";
		delitem getarg(1),1;
		delitem getarg(6),getarg(7);
		if(getarg(0) == 2) delitem getarg(8),getarg(9);
		set Zeny,Zeny - getarg(5)*1000;
		getitem getarg(2),1;
		next;
		show "[Seiyablem]";
		show "Até mais, amigo!";
		close;
	}
	else
	{
		specialeffect EF_FREEZING;
		show "[Seiyablem]";
		show "ahh! ...Desculpa, isso falhou.";
		show "No entanto, não tive culpa.";
		show "É a sua sorte e ela parece ser ruim, ok?";
		show "Não fique desapontado e tente outra vez.";
		delitem getarg(1),1;
		delitem getarg(6),getarg(7);
		if(getarg(0) == 2) delitem getarg(8),getarg(9);
		set Zeny,Zeny - getarg(5)*1000;
		next;
		show "[Seiyablem]";
		show "Desejo boa sorte da próxima vez!";
		close;
	}
}
